## POST /api/v1/organizations/{organization_id}/users

Creates a new user account and immediately adds them to the specified organization. Use this endpoint when you want to create a user and grant them access to an organization in a single operation.

## Request

### Path Parameters

<ParamField path="organization_id" type="string" required>
  Organization ID where the user will be created.
  
  ```
  org_123
  ```
</ParamField>

### Query Parameters

<ParamField query="send_invitation_email" type="boolean">
  If true, sends an activation email to the user. Defaults to true.
  
  ```
  true
  ```
</ParamField>

### Body Parameters

<ParamField body="email" type="string" required>
  Primary email address for the user. Must be unique across the environment.
  
  ```
  user@example.com
  ```
</ParamField>

<ParamField body="external_id" type="string">
  Your application's unique identifier for this user.
  
  ```
  ext_12345a67b89c
  ```
</ParamField>

<ParamField body="user_profile" type="object">
  User's personal information including name and contact details.
  
  <Expandable title="user_profile fields">
    <ParamField body="user_profile.given_name" type="string">
      User's first name. Maximum 255 characters.
    </ParamField>

    <ParamField body="user_profile.family_name" type="string">
      User's last name. Maximum 255 characters.
    </ParamField>

    <ParamField body="user_profile.name" type="string">
      Full name in display format.
    </ParamField>

    <ParamField body="user_profile.phone_number" type="string">
      Phone number in E.164 international format.
      
      ```
      +14155552671
      ```
    </ParamField>

    <ParamField body="user_profile.locale" type="string">
      User's localization preference in BCP-47 format.
      
      ```
      en-US
      ```
    </ParamField>

    <ParamField body="user_profile.picture" type="string">
      URL to the user's profile picture.
    </ParamField>

    <ParamField body="user_profile.custom_attributes" type="object">
      Custom attributes for extended user profile data.
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="membership" type="object">
  Membership details including roles and metadata.
  
  <Expandable title="membership fields">
    <ParamField body="membership.roles" type="array">
      List of roles to assign to the user.
      
      ```json
      [{"name": "member"}]
      ```
    </ParamField>

    <ParamField body="membership.metadata" type="object">
      Custom key-value pairs for storing additional membership context.
      
      ```json
      {
        "department": "engineering",
        "location": "nyc-office"
      }
      ```
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="metadata" type="object">
  Custom key-value pairs for storing additional user context.
  
  ```json
  {
    "department": "engineering",
    "location": "nyc-office"
  }
  ```
</ParamField>

## Response

<ResponseField name="user" type="object">
  The created user object, including system-generated identifiers and timestamps.

  <Expandable title="user object">
    <ResponseField name="id" type="string">
      Unique system-generated identifier for the user
      
      ```
      usr_1234abcd5678efgh
      ```
    </ResponseField>

    <ResponseField name="email" type="string">
      Primary email address for the user
    </ResponseField>

    <ResponseField name="external_id" type="string">
      Your application's unique identifier
    </ResponseField>

    <ResponseField name="user_profile" type="object">
      User's personal information
    </ResponseField>

    <ResponseField name="memberships" type="array">
      List of organization memberships
    </ResponseField>

    <ResponseField name="metadata" type="object">
      Custom key-value pairs
    </ResponseField>

    <ResponseField name="create_time" type="string">
      Timestamp when the user account was created (ISO 8601)
    </ResponseField>

    <ResponseField name="update_time" type="string">
      Timestamp of the last modification (ISO 8601)
    </ResponseField>

    <ResponseField name="last_login_time" type="string">
      Timestamp of the user's most recent successful authentication (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>

    send_invitation_email=True
)

user_id = response[0].user.id

import com.scalekit.ScalekitClient;
