Understand how to handle errors and exceptions in Scalekit SDKs. Proper error handling ensures your application gracefully handles authentication failures, network issues, and invalid requests.

## Common Error Types

Scalekit SDKs return errors in these categories:

| Error Type | Description | Recovery Strategy |
|------------|-------------|------------------|
| **Authentication Errors** | Invalid credentials, expired codes | Redirect to login, refresh tokens |
| **Validation Errors** | Invalid parameters, missing fields | Fix request parameters |
| **Network Errors** | Timeout, connection failures | Retry with exponential backoff |
| **Configuration Errors** | Invalid SDK setup | Check environment variables |
| **Authorization Errors** | Insufficient permissions | Request proper scopes |

## Error Response Format

Errors include these properties:

```json
{
  "error": "invalid_grant",
  "error_description": "Authorization code has expired",
  "status": 400
}
```

| Field | Description |
|-------|-------------|
| `error` | Error code identifier |
| `error_description` | Human-readable error message |
| `status` | HTTP status code |

### Token Expired

Access token has expired:

<Tabs syncKey="tech-stack">
  <TabItem label="Node.js">

```javascript
try {
  const isValid = await scalekit.validateAccessToken(accessToken);

  if (!isValid) {
    // Token expired - try to refresh
    const refreshToken = req.cookies.refreshToken;

    if (refreshToken) {
      const authResult = await scalekit.refreshAccessToken(refreshToken);
      // Update cookies with new tokens
      return authResult.accessToken;

    return res.redirect('/login?error=session_expired');
  }
} catch (error) {
  console.error('Token validation failed:', error);
  return res.status(401).json({ error: 'Invalid token' });
}
```

## Network Errors

### Connection Timeout

Handle network timeouts with retry logic:

<Tabs syncKey="tech-stack">
  <TabItem label="Node.js">
```javascript
async function authenticateWithRetry(code, redirectUri, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await scalekit.authenticateWithCode(code, redirectUri);
    } catch (error) {
      if (error.code === 'ETIMEDOUT' || error.code === 'ECONNREFUSED') {
        if (attempt === maxRetries) {
          console.error('Max retries reached, authentication failed');
          throw new Error('Network error - please try again later');
        }

        // Exponential backoff
        const delay = Math.pow(2, attempt) * 1000;
        console.log(`Attempt ${attempt} failed, retrying in ${delay}ms...`);
        await new Promise(resolve => setTimeout(resolve, delay));
      } else {
        throw error;
      }
    }
  }
}
```
  </TabItem>
  <TabItem label="Python">
```python
import time

def authenticate_with_retry(code, redirect_uri, max_retries=3):
    for attempt in range(1, max_retries + 1):
        try:
            return scalekit_client.authenticate_with_code(
                code, redirect_uri, options
            )
        except Exception as error:
            if 'timeout' in str(error).lower() or 'connection' in str(error).lower():
                if attempt == max_retries:
                    print('Max retries reached, authentication failed')
                    raise Exception('Network error - please try again later')

                # Exponential backoff
                delay = (2 ** attempt)
                print(f'Attempt {attempt} failed, retrying in {delay}s...')
                time.sleep(delay)
            else:
                raise
```
  </TabItem>
  <TabItem label="Go">

```go
import (
    "context"
    "time"
    "math"
)

func authenticateWithRetry(
    ctx context.Context,
    code, redirectUri string,
    maxRetries int,
) (*scalekit.AuthenticationResponse, error) {
    for attempt := 1; attempt <= maxRetries; attempt++ {
        authResult, err := scalekitClient.AuthenticateWithCode(
            ctx, code, redirectUri, scalekit.AuthenticationOptions{},
        )

        if err == nil {
            return authResult, nil
        }

        if isNetworkError(err) {
            if attempt == maxRetries {
                return nil, fmt.Errorf("max retries reached: %w", err)
            }

            // Exponential backoff
            delay := time.Duration(math.Pow(2, float64(attempt))) * time.Second
            log.Printf("Attempt %d failed, retrying in %v...", attempt, delay)
            time.Sleep(delay)
        } else {
            return nil, err
        }
    }

    return nil, fmt.Errorf("authentication failed after %d attempts", maxRetries)
}
```
  </TabItem>
  <TabItem label="Java">
```java
import java.util.concurrent.TimeUnit;

public AuthenticationResponse authenticateWithRetry(
    String code,
    String redirectUri,
    int maxRetries
) throws Exception {
    for (int attempt = 1; attempt <= maxRetries; attempt++) {
        try {
            return scalekitClient
                .authentication()
                .authenticateWithCode(code, redirectUri, new AuthenticationOptions());
        } catch (Exception e) {
            if (isNetworkError(e)) {
                if (attempt == maxRetries) {
                    throw new Exception("Max retries reached: " + e.getMessage());
                }

                // Exponential backoff
                long delay = (long) Math.pow(2, attempt);
                System.out.printf("Attempt %d failed, retrying in %ds...%n", attempt, delay);
                TimeUnit.SECONDS.sleep(delay);
            } else {
                throw e;
            }
        }
    }

    throw new Exception("Authentication failed after " + maxRetries + " attempts");
}

private boolean isNetworkError(Exception e) {
    String message = e.getMessage().toLowerCase();
    return message.contains("timeout") || message.contains("connection");
}
```

## User-Friendly Error Messages

Display helpful error messages to users:

```javascript title="Error mapping"
const ERROR_MESSAGES = {
  invalid_grant: 'Your login session has expired. Please try again.',
  invalid_client: 'Configuration error. Please contact support.',
  access_denied: 'You denied access to your account.',
  network_error: 'Network connection failed. Please check your internet.',
  session_expired: 'Your session has expired. Please sign in again.',
  invalid_state: 'Security validation failed. Please try again.',
};

function getUserMessage(errorCode) {
  return ERROR_MESSAGES[errorCode] || 'An unexpected error occurred. Please try again.';
}

// Usage
try {
  const authResult = await scalekit.authenticateWithCode(code, redirectUri);
} catch (error) {
  const userMessage = getUserMessage(error.code);
  return res.redirect(`/login?error=${encodeURIComponent(userMessage)}`);
}
```

## Best Practices

1. **Always validate input** - Check for required parameters before making SDK calls
2. **Implement retry logic** - Handle transient network errors with exponential backoff
3. **Log errors properly** - Include context for debugging without exposing sensitive data
4. **Provide user feedback** - Show clear, actionable error messages to users
5. **Monitor error rates** - Track error patterns to identify systemic issues
6. **Handle CSRF attacks** - Always validate state parameter in OAuth callbacks
7. **Secure error responses** - Never expose internal errors or stack traces to users
8. **Test error scenarios** - Write tests for common error conditions
